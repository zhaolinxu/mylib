CheckBox全选/反选/至少选1个

全选/反选

$(".check-all").click(function() { if($(this).is(':checked')) { $('input[type="checkbox"]').each(function() { $(this).prop("checked", true); }); } else { $('input[type="checkbox"]').each(function() { $(this).removeAttr("checked", false); }); } });

至少勾选一个才能提交

var num = 0; $("#btn_submit").click(function() { $('input[type="checkbox"]').each(function() { if($(this).is(':checked')) { num += 1; } }); console.log(num); //必须勾选一个 if(num != 0) { alert("至少选一个"); num=0; } else { alert("可以提交"); } });

点击行，勾选

$("tr").click(function(){ //排除全选框 if(!$(this).find("td input[type='checkbox']").is(":checked")){ $(this).find("td input[type='checkbox']").prop("checked",true); }else{ $(this).find("td input[type='checkbox']").prop("checked",false); } //防止点checkbox冒泡 $("input[type='checkbox']").click(function(e) { e.stopPropagation(); }); });
